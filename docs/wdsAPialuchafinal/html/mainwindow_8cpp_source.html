158QVector<QPair<QString, QString>> MainWindow::loadSensorConfig(const QString &filePath)
159{
160 QVector<QPair<QString, QString>> sensors;
161
162 QFile file(filePath);
163 if (!file.open(QIODevice::ReadOnly)) {
164 qWarning() << "Nie można otworzyć pliku:" << filePath;
165 return sensors;
166 }
167
168 QByteArray data = file.readAll();
169 QJsonDocument doc = QJsonDocument::fromJson(data);
170 if (doc.isNull() || !doc.isObject()) {
171 qWarning() << "Niepoprawny JSON";
172 return sensors;
173 }
174
175 QJsonObject rootObj = doc.object();
176 QJsonArray charts = rootObj["charts"].toArray();
177
178 for (const QJsonValue &val : charts) {
179 QJsonObject obj = val.toObject();
180 QString name = obj["sensor"].toString();
181 QString unit = obj["unit_y"].toString();
182 sensors.append(qMakePair(name, unit));
183 }
184
185 return sensors;
186}
