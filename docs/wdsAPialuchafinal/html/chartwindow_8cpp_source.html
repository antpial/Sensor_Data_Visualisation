48QVector<ChartConfig> loadChartConfigs(const QString &filePath)
49{
50 QVector<ChartConfig> configs;
51
52 QFile file(filePath);
53 if (!file.open(QIODevice::ReadOnly)) {
54 qWarning() << "Nie można otworzyć pliku:" << filePath;
55 return configs;
56 }
57
58 QByteArray jsonData = file.readAll();
59 file.close();
60
61 QJsonDocument doc = QJsonDocument::fromJson(jsonData);
62 if (!doc.isObject()) {
63 qWarning() << "Nieprawidłowy format JSON";
64 return configs;
65 }
66
67 QJsonArray chartsArray = doc.object()["charts"].toArray();
68 for (const QJsonValue &value : chartsArray) {
69 QJsonObject obj = value.toObject();
70 ChartConfig config;
71 config.sensorName = obj["sensor"].toString();
72 config.yRange = obj["y_range"].toDouble();
73 config.unitY = obj["unit_y"].toString();
74
75 QString colorStr = obj["color"].toString("#000000"); // domyślnie czarny
76 config.color = QColor(colorStr);
77
78 configs.append(config);
79 }
80
81 return configs;
82}

168void ChartWindow::updateFromPacket(const ParsedPacket &packet)
169{
170 int count = qMin(packet.sensors.size(), seriesList.size());
171
172 for (int i = 0; i < count; ++i) {
173 QLineSeries *series = seriesList[i];
174 QVector<QPointF> points = series->pointsVector();
175
176 if (points.size() >= NUM_OF_SAMPLES)
177 points.removeFirst();
178
179 qreal x = points.isEmpty() ? 0 : points.last().x() + 1;
180 qreal y = packet.sensors[i];
181
182 points.append(QPointF(x, y));
183 series->replace(points);
184
185 QChart *chart = series->chart();
186 if (chart && chart->axisX()) {
187 qreal xMax = points.last().x();
188 qreal xMin = xMax - NUM_OF_SAMPLES;
189 if (xMin < 0) xMin = 0;
190
191 QValueAxis *axisX = qobject_cast<QValueAxis *>(chart->axisX());
192 if (axisX)
193 axisX->setRange(xMin, xMax);
194 }
195 }
